And create `packages/theme/scripts/codegen.mjs`:
```javascript
#!/usr/bin/env node
// Reads packages/theme/src/css/variables.css and emits
// packages/theme/src/tokens/generated.ts as a typed object.

import fs from 'node:fs';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const SRC = path.resolve(__dirname, '..', 'src', 'css', 'variables.css');
const OUT = path.resolve(__dirname, '..', 'src', 'tokens', 'generated.ts');

const css = fs.readFileSync(SRC, 'utf8');
const varRegex = /--([a-z0-9-]+):\s*([^;]+);/gi;
const tokens = {};
let m;
while ((m = varRegex.exec(css))) {
  const key = m[1].replace(/-([a-z])/g, (_, c) => c.toUpperCase());
  tokens[key] = m[2].trim();
}

const lines = [
  '// AUTO-GENERATED by packages/theme/scripts/codegen.mjs — do not hand-edit.',
  '// To regenerate: pnpm --filter @repo/theme codegen',
  '/* eslint-disable */',
  'export const tokens = ' + JSON.stringify(tokens, null, 2) + ' as const;',
  'export type ThemeToken = keyof typeof tokens;',
  '',
];
fs.writeFileSync(OUT, lines.join('\n'));
console.log(`Wrote ${path.relative(process.cwd(), OUT)} (${Object.keys(tokens).length} tokens).`);
```
